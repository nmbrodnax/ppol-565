  map(function(url) read_csv(url, col_types = 'cccccccc')) %>% 
  bind_rows()

PART 2

Assess missing data

tally_missing <- function(df){
  missing <- df %>% 
    mutate_all(is.na) %>% 
    summarise_all(sum) %>% 
    gather(variable, missing)
  return(missing)
}

print(tally_missing(dnc))
